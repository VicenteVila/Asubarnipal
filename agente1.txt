class HybridManager:
    def __init__(self, keys):
        self.keys = keys
        self.current_index = 0
        self.client = genai.Client(api_key=self.keys[self.current_index])
        self.ollama_client = Client(host='http://127.0.0.1:11434')
        self.use_ollama = self._check_ollama()

        # Estado de override para /model
        self.force_gemini = False
        self.force_ollama = False
        self.selected_ollama_model = OLLAMA_MODEL  # Modelo actual
        self.available_ollama_models = []
        self._refresh_ollama_models()

    def _check_ollama(self):
        try: return requests.get("http://127.0.0.1:11434/api/tags", timeout=3).status_code == 200
        except: return False

    def _refresh_ollama_models(self):
        """Actualiza la lista de modelos locales disponibles."""
        try:
            resp = requests.get("http://127.0.0.1:11434/api/tags", timeout=5)
            if resp.status_code == 200:
                data = resp.json()
                self.available_ollama_models = [m.get("name", "") for m in data.get("models", [])]
                log_accion(f"📋 Modelos Ollama detectados: {len(self.available_ollama_models)}")
            else:
                self.available_ollama_models = []
        except Exception as e:
            self.available_ollama_models = []
            log_accion(f"⚠️ Error listando modelos Ollama: {e}")

    def get_active_backend(self):
        """Determina qué backend usar basado en overrides y disponibilidad."""
        if self.force_gemini:
            return "gemini"
        if self.force_ollama and self._check_ollama():
            return "ollama"
        # Default: auto
        if self.use_ollama and self._check_ollama():
            return "ollama"
        return "gemini"

    def set_gemini(self):
        """Forzar modo Gemini."""
        self.force_gemini = True
        self.force_ollama = False
        log_accion("🔵 Backend forzado: Gemini")
        return True

    def set_ollama(self, model_name=None):
        """Forzar modo Ollama, opcionalmente con modelo específico."""
        if not self._check_ollama():
            return False, "Ollama no responde en localhost:11434"

        self.force_ollama = True
        self.force_gemini = False

        if model_name:
            self._refresh_ollama_models()
            if model_name in self.available_ollama_models:
                self.selected_ollama_model = model_name
                log_accion(f"🟢 Backend: Ollama | Modelo: {model_name}")
                return True, f"Ollama activado con modelo: {model_name}"
            else:
                # Modelo no encontrado, listar disponibles
                return False, f"Modelo '{model_name}' no encontrado. Disponibles: {', '.join(self.available_ollama_models[:10])}"
        else:
            # Usar modelo por defecto si está disponible, si no, el primero de la lista
            self._refresh_ollama_models()
            if self.available_ollama_models:
                if self.selected_ollama_model not in self.available_ollama_models:
                    self.selected_ollama_model = self.available_ollama_models[0]
                log_accion(f"🟢 Backend: Ollama | Modelo: {self.selected_ollama_model}")
                return True, f"Ollama activado | Modelo: {self.selected_ollama_model}"
            return False, "No hay modelos Ollama disponibles"

    def list_ollama_models(self):
        """Retorna lista formateada de modelos disponibles."""
        self._refresh_ollama_models()
        if not self.available_ollama_models:
            return "❌ No hay modelos Ollama disponibles. Verifica que Ollama esté corriendo."

        return "\n".join(lines)

async def razonar(prompt, context_type="General", max_tokens=8000):
    """Razona con el LLM híbrido. Ahora incluye el schema en el system prompt."""
    sys_msg = f"""Eres Asubarnipal V18, Arquitecto de Grafos de Conocimiento.

SCHEMA DEL WIKI:
{WIKI_SCHEMA}

REGLAS ADICIONALES:
- Formato: Obsidian Markdown con [[Wikilinks]]
- Contexto: {context_type}
- ANTES de responder, verifica si hay ambigüedad en la solicitud. Si la hay, pregunta.
- SIEMPRE propone la solución más simple primero.
- NUNCA toca código/archivos no relacionados con la tarea actual.
- Define criterios de éxito antes de actuar.
"""
    max_retries = 3
    delay = 2

    # Determinar backend activo
    backend = kernel.get_active_backend()
    ollama_model = kernel.selected_ollama_model

    for attempt in range(max_retries):
        if backend == "ollama":
            try:
                resp = kernel.ollama_client.chat(model=ollama_model, messages=[
                    {'role': 'system', 'content': sys_msg},
                    {'role': 'user', 'content': prompt}
                ])
                return resp.message.content
            except Exception as e:
                log_accion(f"⚠️ Error Ollama (intento {attempt+1}): {e}")
                # Fallback a Gemini si Ollama falla
                backend = "gemini"
                pass

        if backend == "gemini":
            try:
                resp = kernel.client.models.generate_content(model='gemini-2.5-flash', contents=f"{sys_msg}\n\n{prompt}")

                return resp.text
            except Exception as e:
                log_accion(f"⚠️ Reintento {attempt+1} Gemini - Error: {e}")
                kernel.rotate_gemini()
                time.sleep(delay)
                delay *= 2

    return "⚠️ Fallo crítico en el razonamiento."

class WikiEngine:
    """Motor de compilación y mantenimiento del wiki."""

    def __init__(self, obsidian_path: str):
        self.path = Path(obsidian_path)
        self.raw_path = self.path / "raw"
        self.wiki_path = self.path / "wiki"
        self.index_path = self.path / "index.md"
        self.log_path = self.path / "log.md"

        # Asegurar estructura de directorios
        for p in [self.raw_path, self.wiki_path]:
            p.mkdir(parents=True, exist_ok=True)

    def _hash_fuente(self, contenido: str) -> str:
        """Genera hash único para una fuente."""
        return hashlib.md5(contenido.encode()).hexdigest()[:12]

    def _extraer_frontmatter(self, contenido: str) -> Tuple[Dict, str]:
        """Extrae frontmatter YAML de una nota."""
        if contenido.startswith("---"):
            parts = contenido.split("---", 2)
            if len(parts) >= 3:
                try:
                    return yaml.safe_load(parts[1]) or {}, parts[2]
                except Exception:
                    return {}, contenido
        return {}, contenido

    def _buscar_notas_existentes(self) -> List[Dict]:
        """Escanea todas las notas del wiki y retorna metadatos."""
        notas = []
        if not self.wiki_path.exists():
            return notas

        for file_path in self.wiki_path.rglob("*.md"):
            try:
                content = file_path.read_text(encoding="utf-8", errors="ignore")
                fm, body = self._extraer_frontmatter(content)
                notas.append({
                    "id": file_path.stem,
                    "path": str(file_path),
                    "tipo": fm.get("tipo", "unknown"),
                    "titulo": fm.get("titulo", file_path.stem),
                    "tags": fm.get("tags", []),
                    "relacionados": fm.get("relacionados", []),
                    "fecha_ingesta": fm.get("fecha_ingesta", ""),
                    "fuente": fm.get("fuente", "N/A"),
                    "contenido": body[:500],
                })
            except Exception:
                continue
        return notas

    def _encontrar_entidades_relacionadas(self, tema: str, notas: List[Dict]) -> List[Dict]:
        """Busca notas existentes relacionadas con el tema actual."""
        tema_lower = tema.lower()
        relacionadas = []

        for nota in notas:
            score = 0
            # Coincidencia en título
            if tema_lower in nota["titulo"].lower():
                score += 3
            # Coincidencia en tags
            for tag in nota["tags"]:
                if tema_lower in tag.lower():
                    score += 2
            # Coincidencia en contenido
            if tema_lower in nota["contenido"].lower():
                score += 1

            if score > 0:
                relacionadas.append({**nota, "score": score})

        return sorted(relacionadas, key=lambda x: x["score"], reverse=True)[:5]

    def _actualizar_index(self, notas: List[Dict]):
        """Regenera el index.md con todas las notas catalogadas."""
        timestamp = datetime.datetime.now().strftime("%Y-%m-%d %H:%M")

        lines = [
            "# 📇 Index del Wiki Asubarnipal",
            f"",
            f"*Última actualización: {timestamp}*",
            f"",
            "## 📊 Estadísticas",
            f"- Total notas: {len(notas)}",
            f"- Fuentes: {sum(1 for n in notas if n['tipo'] == 'source')}",
            f"- Entidades: {sum(1 for n in notas if n['tipo'] == 'entity')}",
            f"- Conceptos: {sum(1 for n in notas if n['tipo'] == 'concept')}",
            f"- Síntesis: {sum(1 for n in notas if n['tipo'] == 'synthesis')}",
            f"",
            "---",
            f"",
        ]

        # Agrupar por tipo
        por_tipo = {}
        for nota in notas:
            t = nota["tipo"]
            if t not in por_tipo:
                por_tipo[t] = []
            por_tipo[t].append(nota)

        tipo_nombres = {
            "source": "📄 Fuentes",
            "entity": "🏛️ Entidades",
            "concept": "💡 Conceptos",
            "synthesis": "🔗 Síntesis",
            "moc": "🗺️ Mapas de Contenido",
        }

        for tipo, nombre in tipo_nombres.items():
            if tipo in por_tipo:
                lines.append(f"## {nombre}")
                lines.append("")
                for nota in sorted(por_tipo[tipo], key=lambda x: x["titulo"]):
                    lines.append(f"- [[{nota['id']}]] | {nota['titulo']} | Tags: {', '.join(nota['tags'][:3])}")
                lines.append("")

        try:
            with open(self.index_path, "w", encoding="utf-8") as f:
                f.write("\n".join(lines))
            log_accion(f"📇 Index actualizado: {len(notas)} notas catalogadas")
        except Exception as e:
            log_accion(f"⚠️ Error actualizando index: {e}")

    async def ingesta(self, fuente_nombre: str, contenido: str, tipo_fuente: str = "web") -> Dict[str, Any]:
        """
        Pipeline completo de ingesta Karpathy-style:
        1. Guardar fuente cruda
        2. Identificar entidades/conceptos
        3. Actualizar o crear páginas de entidad
        4. Crear nota de fuente
        5. Actualizar index
        """
        timestamp = datetime.datetime.now().strftime("%Y-%m-%d")
        hash_id = self._hash_fuente(contenido)

        # 1. Guardar fuente cruda (inmutable)
        raw_file = self.raw_path / f"{tipo_fuente}_{hash_id}.md"
        try:
            with open(raw_file, "w", encoding="utf-8") as f:
                f.write(f"# Fuente: {fuente_nombre}\n")
                f.write(f"*Tipo: {tipo_fuente} | Hash: {hash_id} | Fecha: {timestamp}*\n\n")
                f.write(contenido)
        except Exception as e:
            log_accion(f"⚠️ Error guardando fuente cruda: {e}")

        # 2. Buscar notas existentes relacionadas
        notas_existentes = self._buscar_notas_existentes()
        relacionadas = self._encontrar_entidades_relacionadas(fuente_nombre, notas_existentes)

        contexto_existente = ""
        if relacionadas:
            contexto_existente = "\n\nNOTAS EXISTENTES RELACIONADAS:\n"
            for nota in relacionadas:
                contexto_existente += f"- [[{nota['id']}]] ({nota['tipo']}): {nota['contenido'][:200]}...\n"

        # 3. Identificar entidades y conceptos clave
        prompt_entidades = f"""Analiza esta fuente y extrae:
1. ENTIDADES concretas (personas, organizaciones, tecnologías, lugares) - máximo 5
2. CONCEPTOS abstractos (teorías, métodos, paradigmas) - máximo 5
3. CONTRADICCIONES con el conocimiento previo (si aplica)

Fuente: {fuente_nombre}
Contenido: {contenido[:8000]}
{contexto_existente}

Responde en formato:
ENTIDADES: nombre1, nombre2, ...
CONCEPTOS: concepto1, concepto2, ...
CONTRADICCIONES: desc1 | nota_relacionada, desc2 | nota_relacionada, ...
SINTESIS_TITULO: Título sugerido para síntesis
"""

        analisis = await razonar(prompt_entidades, "Wiki Ingest Analyzer")

        # Parsear resultado
        entidades = []
        conceptos = []
        contradicciones = []
        sintesis_titulo = fuente_nombre

            if line.startswith("ENTIDADES:"):
                entidades = [e.strip() for e in line.replace("ENTIDADES:", "").split(",") if e.strip()]
            elif line.startswith("CONCEPTOS:"):
                conceptos = [c.strip() for c in line.replace("CONCEPTOS:", "").split(",") if c.strip()]
            elif line.startswith("CONTRADICCIONES:"):
                contras = line.replace("CONTRADICCIONES:", "").split(",")
                contradicciones = [c.strip() for c in contras if c.strip() and c.strip() != "Ninguna"]

                sintesis_titulo = line.replace("SINTESIS_TITULO:", "").strip() or fuente_nombre

        # 4. Actualizar/crear páginas de entidad
        paginas_tocadas = []

        for entidad in entidades:
            entidad_id = f"entity_{entidad.replace(' ', '_').lower()[:30]}"
            entidad_path = self.wiki_path / f"{entidad_id}.md"

            if entidad_path.exists():
                # Actualizar entidad existente
                existing = entidad_path.read_text(encoding="utf-8", errors="ignore")
                fm, body = self._extraer_frontmatter(existing)

                prompt_update = f"""Actualiza esta página de entidad con nueva información.

ENTIDAD: {entidad}
CONTENIDO ACTUAL:
{body[:3000]}

NUEVA INFORMACIÓN DE FUENTE '{fuente_nombre}':
{contenido[:4000]}

CONTRADICCIONES DETECTADAS: {contradicciones}

Instrucciones:
1. Integra la nueva información manteniendo lo que ya estaba
2. SI hay contradicciones, DOCUMENTARLAS explícitamente con fecha
3. Agregar [[{hash_id}]] a relacionados
4. Actualizar fecha_actualizacion
"""
                nuevo_contenido = await razonar(prompt_update, "Entity Updater")

                # Reconstruir con frontmatter actualizado
                nuevo_fm = {
                    **fm,
                    "tipo": "entity",
                    "titulo": entidad,
                    "fecha_actualizacion": timestamp,
                    "fuente": fuente_nombre,
                    "tags": list(set(fm.get("tags", []) + [tipo_fuente])),
                    "relacionados": list(set(fm.get("relacionados", []) + [hash_id])),
                }
            else:
                # Crear nueva entidad
                prompt_create = f"""Crea una página de entidad para: {entidad}

CONTEXTO DE FUENTE:
{contenido[:5000]}

Requisitos:
- Descripción clara y concisa
- Contexto histórico si aplica
- Relación con la fuente [[{hash_id}]]
- 3-5 wikilinks a conceptos relacionados

                    "tipo": "entity",
                    "titulo": entidad,
                    "fecha_ingesta": timestamp,
                    "fecha_actualizacion": timestamp,
                    "fuente": fuente_nombre,
                    "estado": "draft",
                    "tags": [tipo_fuente],
                    "relacionados": [hash_id],
                }

            # Escribir entidad
            header = "---\n" + "\n".join(f"{k}: {json.dumps(v) if isinstance(v, list) else v}" for k, v in nuevo_fm.items()) + "\n---\n\n"
            try:
                with open(entidad_path, "w", encoding="utf-8") as f:
                    f.write(header + nuevo_contenido)
                paginas_tocadas.append(entidad_id)
                log_accion(f"📝 Entidad actualizada: [[{entidad_id}]]")
            except Exception as e:

- TL;DR (3-5 bullets)
- Puntos clave con citas
- Conexiones a entidades: {', '.join([f'[[entity_{e.replace(" ", "_").lower()[:30]}]]' for e in entidades])}
- Preguntas abiertas o gaps
"""
        source_content = await razonar(prompt_source, "Source Summarizer")

        source_fm = {
            "tipo": "source",
            "titulo": fuente_nombre,
            "fecha_ingesta": timestamp,
            "fuente_raw": str(raw_file),
            "estado": "final",
            "tags": [tipo_fuente] + entidades + conceptos,
            "relacionados": paginas_tocadas,
        }

        source_header = "---\n" + "\n".join(f"{k}: {json.dumps(v) if isinstance(v, list) else v}" for k, v in source_fm.items()) + "\n---\n\n"
        try:
            with open(source_path, "w", encoding="utf-8") as f:
                f.write(source_header + source_content)
            paginas_tocadas.append(source_id)
        except Exception as e:
            log_accion(f"⚠️ Error escribiendo fuente {source_id}: {e}")

        # 6. Actualizar index
        todas_notas = self._buscar_notas_existentes()
        self._actualizar_index(todas_notas)

        # 7. Log estructurado
        log_ingest(fuente_nombre, len(paginas_tocadas))

        return {
            "source_id": source_id,
            "entidades": entidades,
            "conceptos": conceptos,
            "contradicciones": contradicciones,
            "paginas_tocadas": paginas_tocadas,
            "relacionadas_previas": [n["id"] for n in relacionadas],
        }

    async def query_wiki(self, pregunta: str) -> str:
        """
        Consulta el wiki existente antes de generar respuesta.
        Lee el index, identifica páginas relevantes, sintetiza con citas.
        """
        notas = self._buscar_notas_existentes()

        if not notas:
            return "📚 El wiki está vacío. Usa /investigar para construir conocimiento primero."

        # Buscar notas relevantes
        pregunta_lower = pregunta.lower()
        relevantes = []
        for nota in notas:
            score = 0
            if any(term in nota["titulo"].lower() for term in pregunta_lower.split()):
                score += 3
            for tag in nota["tags"]:
                if any(term in tag.lower() for term in pregunta_lower.split()):
                    score += 2
            if any(term in nota["contenido"].lower() for term in pregunta_lower.split()):
                score += 1
            if score > 0:
                relevantes.append({**nota, "score": score})

        relevantes = sorted(relevantes, key=lambda x: x["score"], reverse=True)[:5]

        if not relevantes:
            return await razonar(pregunta, "General Query")

        # Construir contexto del wiki
        contexto = "CONTEXTO DEL WIKI:\n\n"
        for nota in relevantes:
            contexto += f"## [[{nota['id']}]] ({nota['tipo']})\n{nota['contenido'][:800]}...\n\n"

        prompt = f"""Responde esta pregunta usando SOLO el contexto del wiki proporcionado.
Si la información no está en el contexto, dilo explícitamente.

PREGUNTA: {pregunta}

{contexto}

Requisitos:
1. Cita las notas del wiki usando [[id]]
2. Si hay contradicciones entre notas, señalarlas
3. Sugerir qué entidades del wiki explorar para profundizar
4. Formato: Markdown con wikilinks
"""

        return await razonar(prompt, "Wiki Query")

    async def lint(self) -> List[str]:
        """
        Health-check del wiki. Detecta problemas de calidad.
        """
        notas = self._buscar_notas_existentes()
        problemas = []

        # 1. Huérfanas (sin links entrantes ni salientes)
        all_ids = {n["id"] for n in notas}
        linked_ids = set()
        for nota in notas:
            for rel in nota.get("relacionados", []):
                linked_ids.add(rel.replace("[[", "").replace("]]", ""))

        for nota in notas:
            if nota["id"] not in linked_ids and not nota.get("relacionados"):
                problemas.append(f"🏝️ HUÉRFANA: [[{nota['id']}]] - Sin conexiones")

        # 2. Stale (sin actualización reciente)
        hoy = datetime.datetime.now()
        for nota in notas:
            try:
                fecha = datetime.datetime.strptime(nota.get("fecha_actualizacion", "2020-01-01"), "%Y-%m-%d")
                dias = (hoy - fecha).days
                if dias > 30 and nota["tipo"] in ["entity", "concept"]:
                    problemas.append(f"⏰ STALE: [[{nota['id']}]] - Última update: {dias} días")
            except Exception:
                pass

        # 3. Sin tags
        for nota in notas:
            if not nota.get("tags"):
                problemas.append(f"🏷️ SIN_TAGS: [[{nota['id']}]]")

        # 4. Contradicciones (buscar palabras clave en contenido)
        for nota in notas:
            if "contradice" in nota["contenido"].lower() or "contradicción" in nota["contenido"].lower():
                problemas.append(f"⚠️ CONTRADICCIÓN: [[{nota['id']}]] - Requiere revisión")

        return problemas

def extract_web(url):
    try:
        r = requests.get(url, timeout=10, headers={'User-Agent': 'Mozilla/5.0'})
        return trafilatura.extract(r.text) or Article(url).text or ""

async def process_ocr_local(file_path: Path) -> str:
    """Llama al modelo GLM-OCR local. 
    Esta función crítica ejecuta la SDK síncrona en un executor 
    para no bloquear el thread asíncrono de Telegram.
    """
    print(f"⚙️ Llamando a GLM-OCR local para reconocimiento de: {file_path.name}")
    
    def sync_ocr_call():
        try:
            # Según el README, debemos instanciar GlmOcr indicando la ruta del config.yaml
            # (Asegúrate de que 'config.yaml' esté en el mismo directorio donde corres el bot)
            with GlmOcr(config_path="config.yaml") as parser:
                
                # El método parse() hace el procesamiento intensivo y la llamada a Ollama
                result = parser.parse(str(file_path))
                
                # El README especifica usar .markdown_result para obtener el texto
                return result.markdown_result
                
        except Exception as e:
            error_msg = f"❌ Error al procesar OCR localmente: {str(e)}"
            print(error_msg)
            return error_msg

    # asyncio.to_thread manda la función síncrona bloqueante a ejecutarse en un hilo separado
    return await asyncio.to_thread(sync_ocr_call)
async def extract_text_from_pdf(file_path: Path) -> str:
    """
    Orquestador de extracción de texto. Intenta pypdf primero, y si falla, usa OCR.
    """
    # 1. INTENTO RÁPIDO (Digital - PyPDF2)
    try:
        # Se recomienda usar la librería "pypdf" en lugar de "PyPDF2" ya que es más moderna.
        import pypdf # Actualizar import si es necesario
        reader = pypdf.PdfReader(file_path)
        text = ""
        
        # Iterar sobre páginas y extraer texto
        for i, page in enumerate(reader.pages):
            text += f"\n[Página {i+1}]\n"
            text += page.extract_text() or "" # Extrae el texto si existe
        
        # Validación de calidad: Si es muy corto, es sospechoso.
        if len(text) < 300 or "Página 1\n[Página 2]" in text: 
            log_accion("⚠️ El texto digital es muy breve o mal formateado. Activando OCR...")
            return await process_ocr_local(file_path)
        
        log_accion("✅ Extracción de texto digital exitosa y suficiente.")
        return text
    except Exception as e:
        # Esto captura el error si PyPDF2 no puede leer la estructura (i.e., es una imagen)
        log_accion(f"❌ Fallo en pypdf ({e}). Asumiendo fuente escaneada. Activando OCR.")
        return await process_ocr_local(file_path)

async def cmd_model(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """
    Comando /model: Gestiona el backend LLM (Gemini vs Ollama).

    Uso:
      /model              → Ver estado actual
      /model gemini       → Forzar Gemini
      /model ollama       → Activar Ollama (modelo por defecto)
      /model ollama <nombre> → Ollama con modelo específico
      /model list         → Listar modelos Ollama disponibles
      /model rotate       → Rotar clave Gemini manualmente
      /model auto         → Volver a modo automático
    """
    args = context.args
    chat_id = update.effective_chat.id

    # Sin argumentos: mostrar estado
    if not args:
        status = kernel.get_status()
        await update.message.reply_text(status, parse_mode="Markdown")
        return

    subcommand = args[0].lower()

    # --- GEMINI ---
    if subcommand == "gemini":
        kernel.set_gemini()
        await update.message.reply_text(
            "🔵 **Gemini activado**\n\n"
            f"• Clave activa: #{kernel.current_index + 1} de {len(kernel.keys)}\n"
            "• Modelo: `gemini-2.5-flash`\n"
            "• Override: ✅ Forzado",
            parse_mode="Markdown"
        )
        return

    # --- OLLAMA ---
    if subcommand == "ollama":
        # Si hay segundo argumento, es el nombre del modelo
        model_name = args[1] if len(args) > 1 else None

        success, message = kernel.set_ollama(model_name)

        if success:
            # Construir mensaje con lista de modelos para selección rápida
            models_text = ""
            if kernel.available_ollama_models:
                models_text = "\n\n📋 **Otros modelos disponibles:**\n"
                for m in kernel.available_ollama_models[:8]:
                    if m != kernel.selected_ollama_model:
                        models_text += f"• `{m}`\n"

            await update.message.reply_text(
                f"🟢 **Ollama activado**\n\n"
                f"• Modelo: `{kernel.selected_ollama_model}`\n"
                f"• Host: `127.0.0.1:11434`\n"
                f"• Total modelos: {len(kernel.available_ollama_models)}"
                f"{models_text}",
                parse_mode="Markdown"
            )
        else:
            await update.message.reply_text(
                f"❌ **Error activando Ollama**\n\n{message}",
                parse_mode="Markdown"
            )
        return

    # --- LIST ---
    if subcommand == "list":
        models_list = kernel.list_ollama_models()
        await update.message.reply_text(models_list, parse_mode="Markdown")
        return

    # --- ROTATE ---
    if subcommand == "rotate":
        old_key = kernel.current_index + 1
        kernel.rotate_gemini()
        new_key = kernel.current_index + 1
        await update.message.reply_text(
            f"🔄 **Clave Gemini rotada**\n\n"
            f"• Anterior: #{old_key}\n"
            f"• Nueva: #{new_key}\n"
            f"• Total claves: {len(kernel.keys)}",
            parse_mode="Markdown"
        )
        return

    # --- AUTO ---
    if subcommand == "auto":
        kernel.force_gemini = False
        kernel.force_ollama = False
        kernel._refresh_ollama_models()
        backend = kernel.get_active_backend()

        await update.message.reply_text(
            "🔄 **Modo automático restaurado**\n\n"
            f"• Backend detectado: `{backend.upper()}`\n"
            f"• Ollama disponible: {'✅' if kernel._check_ollama() else '❌'}\n"
            "• Se usará Ollama si está disponible, Gemini como fallback",
            parse_mode="Markdown"
        )
        return

    # --- AYUDA / DESCONOCIDO ---
    await update.message.reply_text(
        "🤖 **Comando /model**\n\n"
        "Gestiona el backend LLM del agente.\n\n"
        "**Uso:**\n"
        "• `/model` — Estado actual\n"
        "• `/model gemini` — Forzar Gemini\n"
        "• `/model ollama` — Activar Ollama (por defecto)\n"
        "• `/model ollama qwen2.5:7b` — Ollama con modelo específico\n"
        "• `/model list` — Listar modelos locales\n"
        "• `/model rotate` — Rotar clave Gemini\n"
        "• `/model auto` — Modo automático (default)\n\n"
        f"💡 Backend actual: `{kernel.get_active_backend().upper()}`",
        parse_mode="Markdown"
    )


async def cmd_investigar(update: Update, context: ContextTypes.DEFAULT_TYPE):
    topic = " ".join(context.args)
    if not topic: 
        return await update.message.reply_text("❌ Error: Debes indicar un tema. Ej: /investigar IA")

    await update.message.reply_text(f"🌐 Ingestando y compilando en wiki: {topic}...")

    # 1. Búsqueda de información
    try:
        with DDGS() as ddgs:
            urls = [r['href'] for r in ddgs.text(topic, max_results=3)]
    except Exception as e:
        log_accion(f"⚠️ Error en búsqueda: {e}")
        urls = []

    content_full = ""
    for url in urls:
        raw_txt = extract_web(url)
        content_full += f"\n---\nFuente: {url}\n{raw_txt[:4000]}"

    # 2. INGESTA KARPATHY-STYLE
    resultado = await wiki_engine.ingesta(
        fuente_nombre=topic,
        contenido=content_full,
        tipo_fuente="web"
    )

    # 3. Responder al usuario con metadatos de la ingesta
    resumen = f"""✅ Ingesta completada: '{topic}'

📄 Nota de fuente: [[{resultado['source_id']}]]
🏛️ Entidades: {', '.join(resultado['entidades']) if resultado['entidades'] else 'Ninguna nueva'}
💡 Conceptos: {', '.join(resultado['conceptos']) if resultado['conceptos'] else 'Ninguno nuevo'}
⚠️ Contradicciones: {len(resultado['contradicciones'])}
📝 Páginas tocadas: {len(resultado['paginas_tocadas'])}
🔗 Relacionadas previas: {len(resultado['relacionadas_previas'])}

El wiki ha sido actualizado y el index regenerado.
"""

    await update.message.reply_text(resumen)

    # Enviar el archivo de fuente
    source_path = os.path.join(WIKI_PATH, f"{resultado['source_id']}.md")
    if os.path.exists(source_path):
        await update.message.reply_document(open(source_path, 'rb'), 
                                           caption=f"📄 Nota de fuente: {resultado['source_id']}.md")

        )

        context.user_data["charlar_respuesta"] = None
        return

async def handle_charla_libre(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """
    Maneja mensajes durante una sesión de charla libre o socrática.
    Mantiene historial de conversación y contexto.
    """
    msg = update.message.text or ""
    modo_key = context.user_data.get("charlar_mode", "libre")
    tema = context.user_data.get("charlar_tema", "General")
    historial = context.user_data.get("charlar_historial", [])

    # Comandos de salida
    if msg.lower().strip() in ["fin", "salir", "terminar", "exit", "adios", "adiós"]:
        # Resumen de la conversación
        resumen_prompt = f"Resume en 3 bullets la conversación sobre '{tema}':\n"
        for h in historial[-6:]:
            rol = "U" if h["role"] == "user" else "A"
            resumen_prompt += f"{rol}: {h['content'][:100]}\n"

        resumen = await razonar(resumen_prompt, context_type="Resumen")

        keyboard = [
            [InlineKeyboardButton("💾 Guardar resumen en wiki", callback_data="charlar_guardar")],
            [InlineKeyboardButton("🗑️ Solo descartar", callback_data="charlar_descartar")],
        ]

        await update.message.reply_text(
            f"👋 *Sesión finalizada*\n\n"
            f"📋 *Resumen de la conversación:*\n{resumen}\n\n"
            f"¿Quieres guardar esto en el wiki?",
            parse_mode="Markdown",
            reply_markup=InlineKeyboardMarkup(keyboard)
        )

        # Guardar el resumen como respuesta para poder guardarlo
        context.user_data["charlar_respuesta"] = f"# Conversación: {tema}\n\n## Resumen\n{resumen}\n\n## Historial completo\n"
        for h in historial:
            rol = "Usuario" if h["role"] == "user" else "Asistente"
            context.user_data["charlar_respuesta"] += f"**{rol}:** {h['content']}\n\n"

        context.user_data["charlar_mode"] = "esperando_guardar"
        return

    # Añadir mensaje del usuario al historial
    historial.append({"role": "user", "content": msg})

    # Construir prompt con historial (últimos 10 mensajes)
    modo = MODOS_CHARLAR.get(modo_key, MODOS_CHARLAR["libre"])
    historial_text = ""
    for h in historial[-10:]:
        rol = "Usuario" if h["role"] == "user" else "Asistente"
        historial_text += f"{rol}: {h['content']}\n"

    prompt = f"""{modo['prompt_sistema']}

TEMA GENERAL: {tema}

HISTORIAL DE CONVERSACIÓN:
{historial_text}

Responde al último mensaje del usuario manteniendo el tono y estilo del modo activo."""

    respuesta = await razonar(prompt, context_type=modo['nombre'])

    # Guardar respuesta en historial
    historial.append({"role": "assistant", "content": respuesta})
    context.user_data["charlar_historial"] = historial

    # Enviar respuesta (posiblemente en chunks)
    if len(respuesta) > 4000:
        await update.message.reply_text(respuesta[:4000])
        if len(respuesta) > 4000:
            await update.message.reply_text(respuesta[4000:8000])
    else:
        await update.message.reply_text(respuesta)

async def handle_multimedia(update: Update, context: ContextTypes.DEFAULT_TYPE):
    msg = update.message.text or ""

    # === REDIRECCIÓN CHARLA LIBRE / SOCRÁTICO ===
    charlar_mode = context.user_data.get("charlar_mode")
    if charlar_mode in ["libre", "socratico"]:
        return await handle_charla_libre(update, context)

    # === REDIRECCIÓN ESTADO ESPERANDO GUARDAR ===
    if charlar_mode == "esperando_guardar":
        await update.message.reply_text(
            "⏳ Estoy esperando que decidas si guardar la conversación en el wiki.\n"
            "Usa los botones del mensaje anterior o escribe *`fin`* para reiniciar.",
            parse_mode="Markdown"
        )
        return

    # YouTube Hook → Ingesta

    if "youtube.com" in msg or "youtu.be" in msg:
        await update.message.reply_text("📺 Ingestando video en el wiki...")
        try:
            v_id = re.search(r"(?:v=|\/)([0-9A-Za-z_-]{11}).*", msg).group(1)
            srt = YouTubeTranscriptApi.get_transcript(v_id, languages=['es', 'en'])
            txt = " ".join([i['text'] for i in srt])

            resultado = await wiki_engine.ingesta(
                fuente_nombre=f"YouTube: {v_id}",
                contenido=txt[:15000],
                tipo_fuente="youtube"
            )

            resumen = f"""✅ Video ingestado: {v_id}
📄 Nota: [[{resultado['source_id']}]]
🏛️ Entidades: {', '.join(resultado['entidades']) if resultado['entidades'] else 'Ninguna'}
📝 Páginas tocadas: {len(resultado['paginas_tocadas'])}
"""
            await update.message.reply_text(resumen)

        except Exception as e:
            await update.message.reply_text(f"Error YouTube: {e}")

    # Chat General → Query al wiki primero
    elif msg:
        # Intentar responder desde el wiki
        respuesta_wiki = await wiki_engine.query_wiki(msg)

        # Si el wiki no tiene info, fallback a LLM general
        if "vacío" in respuesta_wiki or len(respuesta_wiki) < 50:
            respuesta_wiki = await razonar(msg)

        await update.message.reply_text(respuesta_wiki[:4000])

def imagen_a_ascii(img, ancho=80):
    """Convierte una imagen PIL a arte ASCII."""
    caracteres = "@%#*+=-:. "
    ratio = img.height / img.width
    alto = int(ancho * ratio * 0.5)
    img = img.resize((ancho, alto))
    img = img.convert('L')
    pixeles = img.getdata()
    ascii_str = ""
    for i, pixel in enumerate(pixeles):
        indice = int((pixel / 255) * (len(caracteres) - 1))
        ascii_str += caracteres[indice]
        if (i + 1) % ancho == 0:
            ascii_str += "\n"
    return ascii_str

GRAPH_STORE_PATH = os.path.join(OBSIDIAN_PATH, "graph_store")
EMBEDDING_MODEL = "all-MiniLM-L6-v2"  # ~80MB, rápido, multilingüe
SIMILARITY_THRESHOLD = 0.82  # Umbral para aristas semánticas
TOP_HUBS = 10

# CLASE: WikiVectorIndex
# =============================================================================

class WikiVectorIndex:
    """
    Indexador vectorial del wiki Karpathy.
    Construye grafo híbrido: wikilinks + similitud semántica.
    Detecta comunidades (Louvain) e identifica hubs (betweenness).
    """

    def __init__(self, wiki_path: str, graph_store_path: str):
        self.wiki_path = Path(wiki_path)
        self.graph_store_path = Path(graph_store_path)
        self.graph_store_path.mkdir(parents=True, exist_ok=True)

        self.notes: List[Dict] = []
        self.graph = None
        self.embeddings: Dict[str, np.ndarray] = {}
        self.model = None
        self.comunidades: Dict[str, int] = {}
        self.hubs: List[Tuple[str, float]] = []

        if HAS_SENTENCE_TRANSFORMERS:
            try:
                self.model = SentenceTransformer(EMBEDDING_MODEL)
                log_accion(f"🧠 Modelo de embeddings cargado: {EMBEDDING_MODEL}")
            except Exception as e:
                log_accion(f"⚠️ Error cargando modelo: {e}")
                self.model = None

    def _extraer_frontmatter(self, contenido: str) -> Tuple[Dict, str]:
        if contenido.startswith("---"):
            parts = contenido.split("---", 2)
            if len(parts) >= 3:
                try:
                    return yaml.safe_load(parts[1]) or {}, parts[2]
                except yaml.YAMLError:
                    return {}, contenido
        return {}, contenido

    def escanear_wiki(self) -> List[Dict]:
        """Escanea todas las notas del wiki."""
        notas = []
        if not self.wiki_path.exists():
            return notas

        for file_path in self.wiki_path.rglob("*.md"):
            try:
                content = file_path.read_text(encoding="utf-8", errors="ignore")
                fm, body = self._extraer_frontmatter(content)

                # Extraer wikilinks del cuerpo
                wikilinks = re.findall(r"\[\[([^\]]+)\]\]", body)

                notas.append({
                    "id": file_path.stem,
                    "path": str(file_path),
                    "tipo": fm.get("tipo", "unknown"),
                    "titulo": fm.get("titulo", file_path.stem),
                    "tags": fm.get("tags", []),
                    "relacionados": fm.get("relacionados", []),
                    "estado": fm.get("estado", "unknown"),
                    "content": body[:2000],  # Para embedding
                    "full_content": body,
                    "wikilinks": wikilinks,
                    "word_count": len(body.split()),
                })
            except Exception:
                continue

        self.notes = notas
        log_accion(f"📄 Wiki escaneado: {len(notas)} notas")
        return notas

    def generar_embeddings(self) -> Dict[str, np.ndarray]:
        """Genera embeddings semánticos para cada nota."""
        if not self.model or not HAS_NUMPY:
            log_accion("⚠️ Embeddings no disponibles (falta modelo o numpy)")
            return {}

        texts = []
        ids = []
        for note in self.notes:
            # Combinar título, tags y contenido para embedding rico
            text = f"{note['titulo']}. {' '.join(note['tags'])}. {note['content'][:500]}"
            texts.append(text)
            ids.append(note["id"])

        if not texts:
            return {}

        log_accion(f"🧠 Generando embeddings para {len(texts)} notas...")
        try:
            embeddings_array = self.model.encode(texts, show_progress_bar=False)
            self.embeddings = {ids[i]: embeddings_array[i] for i in range(len(ids))}
            log_accion(f"✅ Embeddings generados: {len(self.embeddings)}")
            return self.embeddings
        except Exception as e:
            log_accion(f"⚠️ Error generando embeddings: {e}")
            return {}

    def construir_grafo(self) -> Optional[Any]:
        """Construye grafo híbrido: wikilinks + similitud semántica."""
        if not HAS_NETWORKX:
            log_accion("⚠️ networkx no disponible. Grafo no construido.")
            return None

        G = nx.DiGraph()

        # 1. Añadir nodos
        for note in self.notes:
            G.add_node(note["id"], 
                      tipo=note["tipo"],
                      titulo=note["titulo"],
                      tags=note["tags"],
                      estado=note["estado"],
                      word_count=note["word_count"])

        # 2. Aristas por wikilinks
        aristas_wikilinks = 0
        for note in self.notes:
            for link in note["wikilinks"]:
                target = link.split("|")[0].strip()  # Manejar alias
                target_clean = target.replace(" ", "_").lower()

                # Buscar nodo objetivo (puede ser con o sin prefijo entity_/concept_)
                target_id = None
                for node_id in G.nodes():
                    if node_id.lower() == target_clean or node_id.lower().endswith(target_clean):
                        target_id = node_id
                        break

                if target_id and target_id != note["id"]:
                    G.add_edge(note["id"], target_id, 
                              tipo="wikilink", 
                              peso=1.0)
                    aristas_wikilinks += 1

        log_accion(f"🔗 Aristas wikilinks: {aristas_wikilinks}")

        # 3. Aristas por similitud semántica
        aristas_semanticas = 0
        if self.embeddings and HAS_SKLEARN and HAS_NUMPY:
            ids = list(self.embeddings.keys())
            vectors = np.array([self.embeddings[i] for i in ids])

            # Calcular matriz de similitud
            sim_matrix = cosine_similarity(vectors)

            for i in range(len(ids)):
                for j in range(i + 1, len(ids)):
                    sim = sim_matrix[i][j]
                    if sim > SIMILARITY_THRESHOLD:
                        G.add_edge(ids[i], ids[j],
                                  tipo="semantica",
                                  peso=float(sim))
                        aristas_semanticas += 1

        log_accion(f"🧠 Aristas semánticas (>{SIMILARITY_THRESHOLD}): {aristas_semanticas}")

        self.graph = G
        return G

    def detectar_comunidades(self) -> Dict[str, int]:
        """Detecta comunidades temáticas con Louvain."""
        if not self.graph or not HAS_NETWORKX:
            return {}

        try:
            # Convertir a grafo no dirigido para Louvain
            G_undirected = self.graph.to_undirected()

            # Louvain community detection
            communities = nx.community.louvain_communities(G_undirected, seed=42)

            # Asignar ID de comunidad a cada nodo
            comunidad_map = {}
            for comm_id, comm_nodes in enumerate(communities):
                for node in comm_nodes:
                    comunidad_map[node] = comm_id

            self.comunidades = comunidad_map
            log_accion(f"📊 Comunidades detectadas: {len(communities)}")

            # Log de comunidades
            for comm_id, nodes in enumerate(communities):
                tipos = Counter(self.graph.nodes[n].get("tipo", "unknown") for n in nodes)
                log_accion(f"   Comunidad {comm_id}: {len(nodes)} nodos ({dict(tipos)})")

            return comunidad_map

        except Exception as e:
            log_accion(f"⚠️ Error detectando comunidades: {e}")
            return {}

    def identificar_hubs(self) -> List[Tuple[str, float]]:
        """Identifica hubs por betweenness centrality."""
        if not self.graph or not HAS_NETWORKX:
            return []

        try:
            # Betweenness centrality (qué nodos actúan de puentes)
            betweenness = nx.betweenness_centrality(self.graph.to_undirected())

            # Degree centrality (qué nodos son más conectados)
            degree_cent = nx.degree_centrality(self.graph.to_undirected())

            # Combinar métricas
            hub_score = {}
            for node in self.graph.nodes():
                b = betweenness.get(node, 0)
                d = degree_cent.get(node, 0)
                hub_score[node] = (b * 0.6) + (d * 0.4)  # Ponderar betweenness

            # Top hubs
            self.hubs = sorted(hub_score.items(), key=lambda x: x[1], reverse=True)[:TOP_HUBS]
            log_accion(f"🏛️ Hubs identificados: {len(self.hubs)}")

            for node, score in self.hubs:
                tipo = self.graph.nodes[node].get("tipo", "?")
                log_accion(f"   {node} ({tipo}): {score:.4f}")

            return self.hubs

        except Exception as e:
            log_accion(f"⚠️ Error identificando hubs: {e}")
            return []

    def persistir(self) -> bool:
        """Persiste grafo + embeddings + metadatos en disco."""
        try:
            # 1. Guardar grafo (NetworkX)
            if self.graph and HAS_NETWORKX:
                graph_path = self.graph_store_path / "wiki_graph.pkl"
                with open(graph_path, "wb") as f:
                    pickle.dump(self.graph, f)
                log_accion(f"💾 Grafo persistido: {graph_path}")

            # 2. Guardar embeddings
            if self.embeddings and HAS_NUMPY:
                emb_path = self.graph_store_path / "embeddings.pkl"
                with open(emb_path, "wb") as f:
                    pickle.dump(self.embeddings, f)
                log_accion(f"💾 Embeddings persistidos: {emb_path}")

            # 3. Guardar metadatos (comunidades, hubs, stats)
            meta = {
                "total_notas": len(self.notes),
                "total_nodos": len(self.graph.nodes()) if self.graph else 0,
                "total_aristas": len(self.graph.edges()) if self.graph else 0,
                "comunidades": self.comunidades,
                "hubs": self.hubs,
                "timestamp": datetime.datetime.now().isoformat(),
            }
            meta_path = self.graph_store_path / "metadata.json"
            with open(meta_path, "w", encoding="utf-8") as f:
                json.dump(meta, f, indent=2, default=str)
            log_accion(f"💾 Metadatos persistidos: {meta_path}")

            # 4. Generar reporte visual en markdown
            self._generar_reporte()

            return True

        except Exception as e:
            log_accion(f"❌ Error persistiendo: {e}")
            return False

    def _generar_reporte(self):
        """Genera reporte markdown del grafo para lectura humana."""
        report_path = self.graph_store_path / "graph_report.md"

        lines = [
            "# 🕸️ Reporte del Grafo Vectorial Asubarnipal",
            "",
            f"*Generado: {datetime.datetime.now().strftime('%Y-%m-%d %H:%M:%S')}*",
            "",
            "## 📊 Estadísticas",
            f"- Total notas indexadas: {len(self.notes)}",
            f"- Nodos en grafo: {len(self.graph.nodes()) if self.graph else 0}",
            f"- Aristas totales: {len(self.graph.edges()) if self.graph else 0}",
            f"- Comunidades detectadas: {len(set(self.comunidades.values())) if self.comunidades else 0}",
            f"- Embeddings generados: {len(self.embeddings)}",
            "",
            "## 🏛️ Hubs Centrales (Top 10)",
            "",
        ]

        for i, (node, score) in enumerate(self.hubs, 1):
            tipo = self.graph.nodes[node].get("tipo", "?") if self.graph else "?"
            titulo = self.graph.nodes[node].get("titulo", node) if self.graph else node
            lines.append(f"{i}. **[[{node}]]** ({tipo}) — {titulo} — score: {score:.4f}")

        lines.extend(["", "## 📊 Comunidades Temáticas", ""])

        if self.comunidades and self.graph:
            # Agrupar nodos por comunidad
            comm_groups = defaultdict(list)
            for node, comm_id in self.comunidades.items():
                comm_groups[comm_id].append(node)

            for comm_id, nodes in sorted(comm_groups.items()):
                tipos = Counter(self.graph.nodes[n].get("tipo", "unknown") for n in nodes)
                lines.append(f"### Comunidad {comm_id}")
                lines.append(f"- Nodos: {len(nodes)}")
                lines.append(f"- Tipos: {dict(tipos)}")
                lines.append(f"- Miembros: {', '.join([f'[[{n}]]' for n in nodes[:5]])}{'...' if len(nodes) > 5 else ''}")
                lines.append("")

        lines.extend(["", "## 🔗 Aristas Semánticas (Top 20)", ""])

        if self.graph:
            semantic_edges = [(u, v, d) for u, v, d in self.graph.edges(data=True) 
                            if d.get("tipo") == "semantica"]
            semantic_edges.sort(key=lambda x: x[2].get("peso", 0), reverse=True)

            for u, v, d in semantic_edges[:20]:
                peso = d.get("peso", 0)
                lines.append(f"- [[{u}]] ↔ [[{v}]] (similitud: {peso:.3f})")

        with open(report_path, "w", encoding="utf-8") as f:
            f.write("\n".join(lines))

        log_accion(f"📄 Reporte generado: {report_path}")

    def query_semantica(self, query_text: str, top_k: int = 5) -> List[Tuple[str, float]]:
        """Busca notas similares semánticamente a la query."""
        if not self.model or not self.embeddings or not HAS_NUMPY:
            return []

        try:
            query_emb = self.model.encode([query_text])[0]

            results = []
            for note_id, emb in self.embeddings.items():
                sim = float(np.dot(query_emb, emb) / (np.linalg.norm(query_emb) * np.linalg.norm(emb)))
                results.append((note_id, sim))

            results.sort(key=lambda x: x[1], reverse=True)
            return results[:top_k]

        except Exception as e:
            log_accion(f"⚠️ Error en query semántica: {e}")
            return []

    def full_index(self) -> Dict[str, Any]:
        """Pipeline completo: escanear → embed → grafo → comunidades → hubs → persistir."""
        log_accion("🔨 INICIANDO INDEXACIÓN VECTORIAL DEL WIKI")
        log_accion("=" * 50)

        # 1. Escanear
        self.escanear_wiki()
        if not self.notes:
            log_accion("❌ Wiki vacío. Nada que indexar.")
            return {"error": "wiki_vacio"}

        # 2. Embeddings
        self.generar_embeddings()

        # 3. Grafo
        self.construir_grafo()

        # 4. Comunidades
        self.detectar_comunidades()

        # 5. Hubs
        self.identificar_hubs()

        # 6. Persistir
        self.persistir()

        log_accion("=" * 50)
        log_accion("✅ INDEXACIÓN COMPLETADA")

        return {
            "total_notas": len(self.notes),
            "total_nodos": len(self.graph.nodes()) if self.graph else 0,
            "total_aristas": len(self.graph.edges()) if self.graph else 0,
            "embeddings": len(self.embeddings),
            "comunidades": len(set(self.comunidades.values())) if self.comunidades else 0,
            "hubs": len(self.hubs),
        }

async def cmd_indexar_wiki(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Comando /indexar_wiki: construye grafo vectorial del wiki Karpathy con progreso en tiempo real."""
    chat_id = update.effective_chat.id
    start_time = time.time()

    # Mensaje inicial
    progress_msg = await update.message.reply_text("⏳ Iniciando indexación del wiki...")

    # Verificar dependencias
    faltantes = []
    if not HAS_NETWORKX:
        faltantes.append("networkx")
    if not HAS_SENTENCE_TRANSFORMERS:
        faltantes.append("sentence-transformers")
    if not HAS_NUMPY:
        faltantes.append("numpy")
    if not HAS_SKLEARN:
        faltantes.append("scikit-learn")

    if faltantes:
        deps = ", ".join(faltantes)
        await progress_msg.edit_text(f"❌ Dependencias faltantes: {deps}\n\nInstala con: pip install {deps}")
        return

    # Inicializar indexador
    indexer = WikiVectorIndex(WIKI_PATH, GRAPH_STORE_PATH)

    # PASO 1: Escanear wiki
    await progress_msg.edit_text("📄 Escaneando notas del wiki...")
    notas = indexer.escanear_wiki()

    if not notas:
        await progress_msg.edit_text("❌ Wiki vacío. Ejecuta /investigar primero.")
        return

    await progress_msg.edit_text(f"📄 Wiki escaneado: {len(notas)} notas encontradas")
    time.sleep(0.5)

    # PASO 2: Generar embeddings con progreso
    if indexer.model and HAS_NUMPY:
        await progress_msg.edit_text(f"🧠 Cargando modelo de embeddings ({EMBEDDING_MODEL})...")
        # El modelo ya se cargó en __init__, pero puede tardar la primera vez

        total_notas = len(notas)
        embeddings = {}
        texts = []
        ids = []

        for note in notas:
            text = f"{note['titulo']}. {' '.join(note['tags'])}. {note['content'][:500]}"
            texts.append(text)
            ids.append(note["id"])

        # Generar en batches con progreso
        batch_size = 5
        for i in range(0, len(texts), batch_size):
            batch_texts = texts[i:i+batch_size]
            batch_ids = ids[i:i+batch_size]

            try:
                batch_embs = indexer.model.encode(batch_texts, show_progress_bar=False)
                for j, note_id in enumerate(batch_ids):
                    embeddings[note_id] = batch_embs[j]
            except Exception as e:
                log_accion(f"⚠️ Error en batch {i}: {e}")
                continue

            progreso = min(i + batch_size, total_notas)
            porcentaje = int((progreso / total_notas) * 100)
            await progress_msg.edit_text(
                f"🧠 Generando embeddings: {progreso}/{total_notas} ({porcentaje}%)..."
            )
            time.sleep(0.2)  # Pequeña pausa para no saturar Telegram

        indexer.embeddings = embeddings
        await progress_msg.edit_text(f"✅ Embeddings generados: {len(embeddings)}")
        time.sleep(0.5)
    else:
        await progress_msg.edit_text("⚠️ Embeddings no disponibles (modelo no cargado)")

    # PASO 3: Construir grafo
    await progress_msg.edit_text("🔗 Construyendo grafo híbrido (wikilinks + semánticas)...")
    indexer.construir_grafo()

    if indexer.graph:
        aristas_wiki = sum(1 for u, v, d in indexer.graph.edges(data=True) if d.get('tipo') == 'wikilink')
        aristas_sem = sum(1 for u, v, d in indexer.graph.edges(data=True) if d.get('tipo') == 'semantica')
        await progress_msg.edit_text(
            f"🔗 Grafo construido: {len(indexer.graph.nodes())} nodos, "
            f"{aristas_wiki} wikilinks + {aristas_sem} semánticas"
        )
        time.sleep(0.5)

    # PASO 4: Comunidades
    await progress_msg.edit_text("📊 Detectando comunidades temáticas (Louvain)...")
    indexer.detectar_comunidades()

    if indexer.comunidades:
        num_comms = len(set(indexer.comunidades.values()))
        await progress_msg.edit_text(f"📊 Comunidades detectadas: {num_comms}")
        time.sleep(0.5)

    # PASO 5: Hubs
    await progress_msg.edit_text("🏛️ Identificando hubs centrales...")
    indexer.identificar_hubs()

    if indexer.hubs:
        await progress_msg.edit_text(f"🏛️ Hubs identificados: {len(indexer.hubs)}")
        time.sleep(0.5)

    # PASO 6: Persistir
    await progress_msg.edit_text("💾 Persistiendo grafo en disco...")
    indexer.persistir()

    elapsed = int(time.time() - start_time)
    mins, secs = divmod(elapsed, 60)

    # Mensaje final
    resumen = f"""✅ Indexación completada en {mins}m {secs}s

📄 Notas indexadas: {len(notas)}
🕸️ Nodos en grafo: {len(indexer.graph.nodes()) if indexer.graph else 0}
🔗 Aristas totales: {len(indexer.graph.edges()) if indexer.graph else 0}
   ├─ Wikilinks: {sum(1 for u, v, d in indexer.graph.edges(data=True) if d.get('tipo') == 'wikilink') if indexer.graph else 0}
   └─ Semánticas: {sum(1 for u, v, d in indexer.graph.edges(data=True) if d.get('tipo') == 'semantica') if indexer.graph else 0}
🧠 Embeddings: {len(indexer.embeddings)}
📊 Comunidades: {len(set(indexer.comunidades.values())) if indexer.comunidades else 0}
🏛️ Hubs: {len(indexer.hubs)}

💾 Persistido en: {GRAPH_STORE_PATH}
📄 Reporte: graph_report.md

Ahora usa /query_vectorial <pregunta>"""

    await progress_msg.edit_text(resumen)

async def cmd_query_vectorial(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Comando /query_vectorial: busca semánticamente en el wiki indexado."""
    pregunta = " ".join(context.args)
    if not pregunta:
        return await update.message.reply_text("❌ Ejemplo: /query_vectorial ¿Qué es PolarQuant?")

    # Verificar que existe index
    graph_store = Path(GRAPH_STORE_PATH)
    if not (graph_store / "embeddings.pkl").exists():
        return await update.message.reply_text("❌ Wiki no indexado. Ejecuta /indexar_wiki primero.")

    await update.message.reply_text(f"🔍 Buscando: '{pregunta}'...")

    try:
        # Cargar index
        indexer = WikiVectorIndex(WIKI_PATH, GRAPH_STORE_PATH)

        # Cargar embeddings persistidos
        with open(graph_store / "embeddings.pkl", "rb") as f:
            indexer.embeddings = pickle.load(f)

        # Cargar modelo
        if HAS_SENTENCE_TRANSFORMERS:
            indexer.model = SentenceTransformer(EMBEDDING_MODEL)

        # Query
        resultados = indexer.query_semantica(pregunta, top_k=5)

        if not resultados:
            return await update.message.reply_text("🔍 No se encontraron resultados.")

        respuesta = "🔍 RESULTADOS SEMÁNTICOS:\n\n"
        for i, (note_id, sim) in enumerate(resultados, 1):
            # Buscar título de la nota
            titulo = note_id
            for note in indexer.notes:
                if note["id"] == note_id:
                    titulo = note.get("titulo", note_id)
                    break

            respuesta += f"{i}. **[[{note_id}]]** ({titulo})\n"
            respuesta += f"   Similitud: {sim:.3f}\n\n"

        await update.message.reply_text(respuesta)

    except Exception as e:
        await update.message.reply_text(f"⚠️ Error en query: {e}")
